        %% Transitions from Running (Initiation of Shutdown)
        Running -- "Remove(kPackage) on Boot Driver" --> Prestop
        Running -- "Remove(kAll) OR<br>Remove(kPackage) on Package Driver" --> WaitingOnBind
        Prestop -- "Remove(kAll)" --> WaitingOnBind

        %% Main State Machine Flow
        WaitingOnBind --> C1{IsPendingBind is false} --> WaitingOnChildren
        WaitingOnChildren --> C2{HasChildren is false} --> A2[/Action: StopDriver/] --> WaitingOnDriver
        WaitingOnDriver --> C3{HasDriver is false<br>or<br>Driver FIDL closed} --> A3[/Action: StopDriverComponent/] --> WaitingOnComponent
        WaitingOnComponent --> C4{HasDriverComponent is false<br>or<br>Runner controller closed} --> Stopped
        Stopped --> A4[/Action: MaybeDestroyDriverComponent/]
        A4 -- "returns true" --> WaitingOnDestroy
        Stopped --> C5{StartDriver called} --> A5[/Action: ResetShutdown/]
        WaitingOnDestroy --> C6{HasDriverComponentController is false<br>or<br>Controller closed} --> A6[/Action: FinishShutdown/]

        A6 --> Destroyed

        %% Finalization Flow
        Destroyed -- "Intent: kRemoval / kRebindComposite" --> Destructor
        Destroyed -- "Intent: kRestart / kQuarantine" --> A5[/Action: ResetShutdown/]

        A5[/Action: ResetShutdown/] --> Running
    end
